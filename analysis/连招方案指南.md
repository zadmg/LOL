### 0.3 通用技能预判函数

```lua
-- 从 LeagueSharp.Common/Prediction.cs 移植
function GetPrediction(target, spell)
    -- spell = {Range=, Speed=, Delay=, Width=, Type=Line/Circle/Cone, Collision=true}
    local targetPos = target.pos
    local targetVel = target.pathing.vel  -- 速度向量
    local dist = GetDistance(myHero.pos, targetPos)

    -- 计算命中时间(弹道到达时间)
    local hitTime = spell.Delay + (dist / spell.Speed)

    -- 预测目标位置
    local predictedPos = targetPos + targetVel * hitTime

    -- 碰撞检测(如果需要)
    if spell.Collision then
        if IsCollision(myHero.pos, predictedPos, spell.Width, "minion") then
            return nil, 0  -- 被小兵阻挡,不施放
        end
    end

    -- 命中概率
    local hitChance = 4  -- Medium
    if target.isStunned or target.isRooted then
        hitChance = 8  -- Immobile,必中
    elseif target.isDashing then
        hitChance = 7  -- Dashing,高命中
    end

    return predictedPos, hitChance
end
```

```lua
function Combo()
    local target = ts.target
    if not ValidTarget(target, 1200) then return end

    -- Q: 当有 4 层专注时激活(普攻增强)
    if Config.ComboSub.useQ and myHero:CanUseSpell(_Q) == READY then
        if GetBuffStacks(myHero, "AsheQAttack") >= 4 then
            CastSpell(_Q)
        end
    end

    -- W: 减速目标,与 Q 配合
    if Config.ComboSub.useW and myHero:CanUseSpell(_W) == READY then
        local pos, hc = GetPrediction(target, SpellW)
        if pos and hc >= 4 and GetDistance(pos) <= SpellW.Range then
            CastSpell(_W, pos.x, pos.z)
        end
    end

    -- R: 全图先手(默认手动,可设为低血量目标)
    if Config.ComboSub.useR and myHero:CanUseSpell(_R) == READY then
        if target.health < target.maxHealth * 0.4 then  -- 残血目标
            local pos, hc = GetPrediction(target, SpellR)
            if pos and hc >= 6 then  -- 高命中概率
                CastSpell(_R, pos.x, pos.z)
            end
        end
    end
end
```

### 3.2 E 伤害公式(直接替换辅助中 0x1800199E0 表的哈希查找)

```lua
function eDmgCalc(unit, stacks)
    local first = {dmg={20,30,40,50,60}, scaling=0.60}
    local adds  = {dmg={5,9,14,20,27}, scaling={0.15,0.18,0.21,0.24,0.27}}

    local mainDmg  = first.dmg[level] + first.scaling * myHero.totalDamage
    local extraDmg = (stacks > 1) and
                     (adds.dmg[level] + adds.scaling[level] * myHero.totalDamage) * (stacks - 1) or 0
    return myHero:CalcDamage(unit, mainDmg + extraDmg)
end
```

### 3.3 连招决策(已完整实现,见 5.3.7.2 节)

- Q: 目标在 1200 范围 + 无小兵阻挡 → 预判施放
- E(自动斩杀): 层数 >= 阈值 → 施放
- E(超范围提前): 预测目标即将离开 1000 范围 → 提前释放
- E(打野怪): 检测野怪营地,血量 < E 伤害 → 抢野

```lua
-- R 伤害(根据目标血量加成)

    local base = {70, 110, 150}[myHero:GetSpellData(_R).level]

    local multiplier = 1 + math.min(missingRatio * 2, 1.0)  -- 最高 2 倍
    return myHero:CalcMagicDamage(unit, base * multiplier + 0.35 * myHero.ap)
```

### 5.2 连招逻辑

```lua
function Combo()
    local target = ts.target
    if not ValidTarget(target, 1000) then return end

    -- Q: 回旋镖(预判目标走位,考虑来回伤害)
    if Config.ComboSub.useQ and myHero:CanUseSpell(_Q) == READY then
        local pos, hc = GetPrediction(target, SpellQ)
        if pos and hc >= 4 and GetDistance(pos) <= SpellQ.Range then
            CastSpell(_Q, pos.x, pos.z)
        end
    end

    -- W: 弹射(普攻时开启)
    if Config.ComboSub.useW and myHero:CanUseSpell(_W) == READY then
        if not HasBuff(myHero, "SivirW") then
            CastSpell(_W)
        end
    end

    -- E: 自动格挡敌方技能
    if Config.ComboSub.useE and myHero:CanUseSpell(_E) == READY then
        -- 检测敌方 skillshot 即将命中
        for _, enemy in ipairs(GetEnemyHeroes()) do
            if IsValidTarget(enemy, 1000) then
                local spell = GetIncomingSpell(myHero, enemy)
                if spell and spell.time < 0.3 then  -- 0.3 秒内命中
                    CastSpell(_E)
                    break
                end
            end
        end
    end

    -- R: 团队加速(队友接近时使用)
    if Config.ComboSub.useR and myHero:CanUseSpell(_R) == READY then
        local alliesNear = CountAlliesInRange(myHero, 1000)
        if alliesNear >= 2 then  -- 有 2 个以上队友
            CastSpell(_R)
        end
    end
end
```

```lua
function Combo()

    if not ValidTarget(target, 900) then return end

    -- Q: 开启攻速(进入战斗时)
    if Config.ComboSub.useQ and myHero:CanUseSpell(_Q) == READY then
        if GetDistance(target) < 550 then  -- 普攻范围
            CastSpell(_Q)
        end
    end

    -- E: 给目标上爆破电荷
    if Config.ComboSub.useE and myHero:CanUseSpell(_E) == READY then
        if ValidTarget(target, SpellE.Range) and not HasBuff(target, "TristanaECharge") then
            CastSpell(_E, target)
        end
    end

    -- W: 跳跃接近(W 击杀重置)

        local dist = GetDistance(target)
        if dist > 550 and dist < 900 then  -- 普攻打不到,W 能跳
            CastSpell(_W, target.x, target.z)
        end
    end

    -- R: 斩杀+击退
    if Config.ComboSub.useR and myHero:CanUseSpell(_R) == READY then
        if ValidTarget(target, SpellR.Range) then
            local dmg = RDamage(target)
            if target.health < dmg then
                CastSpell(_R, target)  -- 击杀
            elseif CountEnemyHeroesInRange(myHero, 400) >= 2 then
                CastSpell(_R, target)  -- 自保击退
```

```lua
        local dist = GetDistance(target)
        if dist > 700 and dist < 1200 then
            CastSpell(_Q)  -- 隐身接近
        end
    end

    -- W: 叠加毒层(预判目标走位)
    if Config.ComboSub.useW and myHero:CanUseSpell(_W) == READY then
        local pos, hc = GetPrediction(target, SpellW)
        if pos and hc >= 4 and GetDistance(pos) <= SpellW.Range then
            CastSpell(_W, pos.x, pos.z)
        end
    end

    -- R: 开启攻击力+射程(进入战斗时)
    if Config.ComboSub.useR and myHero:CanUseSpell(_R) == READY then
        if not HasBuff(myHero, "TwitchR") and GetDistance(target) < 600 then
            CastSpell(_R)
        end
    end

    -- E: 毒层斩杀(自动触发)
    if Config.ComboSub.useE and myHero:CanUseSpell(_E) == READY then
        if ValidTarget(target, SpellE.Range) then
            local stacks = GetBuffStacks(target, "TwitchDeadlyVenom")
            if stacks >= 4 then
                local dmg = EDamage(target, stacks)
                if target.health < dmg then  -- 可击杀
                    CastSpell(_E, target)
                elseif stacks == 6 then  -- 满层直接释放
                    CastSpell(_E, target)
                end
            end
        end
    end
end
```

### 8.3 伤害公式

```lua
-- E 毒性爆发(基于毒层数)
function EDamage(unit, stacks)
    local base = {20, 35, 50, 65, 80}[myHero:GetSpellData(_E).level]
    local perStack = {10, 15, 20, 25, 30}[myHero:GetSpellData(_E).level]
    return myHero:CalcMagicDamage(unit, base + perStack * stacks + 0.30 * myHero.ap + 0.25 * myHero.totalDamage)
end
```

```lua
function Combo()

    -- Q: 双刃消耗
    if Config.ComboSub.useQ and myHero:CanUseSpell(_Q) == READY then

        end
    end

    -- W: 攻速开启(进入普攻范围时)
    if Config.ComboSub.useW and myHero:CanUseSpell(_W) == READY then
        if GetDistance(target) < 525 and not HasBuff(myHero, "XayahW") then
            CastSpell(_W)
        end
    end

            CastSpell(_E)
        elseif target.health < EDamage(target, #feathers) then  -- 可击杀
            CastSpell(_E)
        end
    end

    -- R: 不可选中(敌方 skillshot 即将命中)
    if Config.ComboSub.useR and myHero:CanUseSpell(_R) == READY then
        -- 检测高危技能即将命中
        local dangerSpells = {"BlitzcrankQ", "ThreshQ", "MorganaQ", "NautilusQ"}
        for _, spellName in ipairs(dangerSpells) do
            local incoming = GetIncomingSpell(myHero, spellName)
            if incoming and incoming.time < 0.3 then
                CastSpell(_R)
                break
            end
        end
    end
end
```

### 9.3 伤害公式

```lua
-- E 致死羽芒(基于穿过目标的羽毛数)
function EDamage(unit, featherCount)
    local base = {50, 60, 70, 80, 90}[myHero:GetSpellData(_E).level]
    local perFeather = 0.6  -- 后续羽毛伤害递减
    local dmg = base + 0.60 * myHero.totalDamage
    -- 穿过的每根羽毛伤害递减(第 1 根 100%,第 2 根 60%,第 3 根 36%,...)
    for i = 1, featherCount do
        dmg = dmg + (base + 0.60 * myHero.totalDamage) * math.pow(perFeather, i - 1)
    end
    return myHero:CalcMagicDamage(unit, math.min(dmg, base * 10))  -- 上限 10 倍
end
```
